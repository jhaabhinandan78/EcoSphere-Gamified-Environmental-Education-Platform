rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ─── Helper Functions ───────────────────────────────────────────
    
    function isSignedIn() {
      return request.auth != null;
    }
    
    function getCurrentUser() {
      return get(/databases/$(database)/documents/Users/$(request.auth.uid)).data;
    }
    
    function isSameSchool(schoolId) {
      return isSignedIn() && getCurrentUser().get('schoolId', '') == schoolId;
    }
    
    function isLeadTeacher() {
      return isSignedIn() && getCurrentUser().get('isLeadTeacher', false) == true;
    }
    
    function isTeacher() {
      return isSignedIn() && getCurrentUser().role == "teacher";
    }
    
    function isApprovedTeacher() {
      return isTeacher() && getCurrentUser().get('isApproved', false) == true;
    }
    
    function isStudent() {
      return isSignedIn() && getCurrentUser().role == "student";
    }
    
    function isAdmin() {
      return isSignedIn() && getCurrentUser().role == "admin";
    }
    
    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }
    
    // ─── Schools Collection ─────────────────────────────────────────
    
    match /Schools/{schoolId} {
      // Anyone can read schools for code validation during registration
      // This is necessary because users aren't authenticated yet during registration
      allow read: if true;
      
      // Anyone can create schools (for web portal registration)
      // This allows the school registration portal to create new schools
      allow create: if true;
      
      // Only admins can update and delete schools
      allow update, delete: if isAdmin();
    }
    
    // ─── Users Collection ───────────────────────────────────────────
    
    match /Users/{userId} {
      // Anyone can read users for first teacher check during registration
      // This is necessary because users aren't authenticated yet during registration
      // Limited to checking if teachers exist for a school
      allow read: if true;
      
      // Users can read their own data
      allow read: if isSignedIn() && request.auth.uid == userId;
      
      // Users can update their own data (except role, schoolId, isApproved)
      allow update: if isSignedIn() && 
                       request.auth.uid == userId &&
                       request.resource.data.role == resource.data.role &&
                       request.resource.data.get('schoolId', '') == resource.data.get('schoolId', '') &&
                       request.resource.data.get('isApproved', false) == resource.data.get('isApproved', false);
      
      // Approved teachers can read students from their school
      allow read: if isSignedIn() && 
                     isApprovedTeacher() &&
                     resource.data.get('schoolId', '') != '' &&
                     isSameSchool(resource.data.schoolId) &&
                     resource.data.role == "student";
      
      // Lead Teacher can approve teachers from their school
      allow update: if isSignedIn() &&
                       isLeadTeacher() &&
                       resource.data.get('schoolId', '') != '' &&
                       isSameSchool(resource.data.schoolId) &&
                       resource.data.role == "teacher" &&
                       request.resource.data.role == resource.data.role &&
                       request.resource.data.get('schoolId', '') == resource.data.get('schoolId', '') &&
                       request.resource.data.get('isLeadTeacher', false) == resource.data.get('isLeadTeacher', false);
      
      // Anyone can create their own user (registration)
      allow create: if request.auth.uid == userId;
      
      // Admins can read all users
      allow read: if isAdmin();
      
      // Admins can update user data
      allow update: if isAdmin();
      
      match /{subcollection=**} {
        allow read: if isOwner(userId) || isAdmin();
        allow write: if isOwner(userId) || isAdmin();
      }
    }
    
    // ─── Batches Collection ─────────────────────────────────────────
    
    match /Batches/{batchId} {
      // Anyone can read batches for registration (students need to see available batches)
      // This is necessary because users aren't authenticated yet during registration
      allow read: if true;
      
      // Approved teachers and Lead Teachers can create batches for their school
      allow create: if isSignedIn() && 
                       isTeacher() &&
                       request.resource.data.schoolId == getCurrentUser().get('schoolId', '');
      
      // Teachers can update their own batches
      allow update: if isSignedIn() &&
                       isTeacher() &&
                       resource.data.teacherId == request.auth.uid;
      
      // Admins can do everything with batches
      allow read, create, update, delete: if isAdmin();
    }
    
    // ─── Modules Collection ─────────────────────────────────────────
    
    match /Modules/{moduleId} {
      // Teachers and students can read modules from their school
      allow read: if isSignedIn() && 
                     (isAdmin() || 
                      (resource.data.get('schoolId', '') != '' && isSameSchool(resource.data.schoolId)));
      
      // Approved teachers can create modules for their school
      allow create: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         request.resource.data.get('schoolId', '') == getCurrentUser().get('schoolId', '')));
      
      // Approved teachers can update modules from their school
      allow update: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         resource.data.get('schoolId', '') != '' && 
                         isSameSchool(resource.data.schoolId)));
      
      // Approved teachers can delete modules from their school
      allow delete: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         resource.data.get('schoolId', '') != '' && 
                         isSameSchool(resource.data.schoolId)));
      
      // Questions subcollection
      match /questions/{questionId} {
        // Inherit parent module's schoolId for access control
        allow read: if isSignedIn() && 
                       (isAdmin() || 
                        (exists(/databases/$(database)/documents/Modules/$(moduleId)) &&
                         get(/databases/$(database)/documents/Modules/$(moduleId)).data.get('schoolId', '') != '' &&
                         isSameSchool(get(/databases/$(database)/documents/Modules/$(moduleId)).data.schoolId)));
        
        allow write: if isSignedIn() && 
                        (isAdmin() || 
                         (isApprovedTeacher() &&
                          exists(/databases/$(database)/documents/Modules/$(moduleId)) &&
                          get(/databases/$(database)/documents/Modules/$(moduleId)).data.get('schoolId', '') != '' &&
                          isSameSchool(get(/databases/$(database)/documents/Modules/$(moduleId)).data.schoolId)));
      }
    }
    
    // ─── Challenges Collection ──────────────────────────────────────
    
    match /Challenges/{challengeId} {
      // Teachers and students can read challenges from their school
      allow read: if isSignedIn() && 
                     (isAdmin() || 
                      (resource.data.get('schoolId', '') != '' && isSameSchool(resource.data.schoolId)));
      
      // Approved teachers can create challenges for their school
      allow create: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         request.resource.data.get('schoolId', '') == getCurrentUser().get('schoolId', '')));
      
      // Approved teachers can update challenges from their school
      allow update: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         resource.data.get('schoolId', '') != '' && 
                         isSameSchool(resource.data.schoolId)));
      
      // Approved teachers can delete challenges from their school
      allow delete: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         resource.data.get('schoolId', '') != '' && 
                         isSameSchool(resource.data.schoolId)));
    }
    
    // ─── Quizzes Collection ─────────────────────────────────────────
    
    match /Quizzes/{quizId} {
      // Teachers and students can read quizzes from their school
      allow read: if isSignedIn() && 
                     (isAdmin() || 
                      (resource.data.get('schoolId', '') != '' && isSameSchool(resource.data.schoolId)));
      
      // Approved teachers can create quizzes for their school
      allow create: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         request.resource.data.get('schoolId', '') == getCurrentUser().get('schoolId', '')));
      
      // Approved teachers can update quizzes from their school
      allow update: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         resource.data.get('schoolId', '') != '' && 
                         isSameSchool(resource.data.schoolId)));
      
      // Approved teachers can delete quizzes from their school
      allow delete: if isSignedIn() && 
                       (isAdmin() || 
                        (isApprovedTeacher() && 
                         resource.data.get('schoolId', '') != '' && 
                         isSameSchool(resource.data.schoolId)));
    }
    
    // ─── Challenge Submissions ──────────────────────────────────────
    
    match /ChallengeSubmissions/{submissionId} {
      // Students can read their own submissions
      allow read: if isSignedIn() && 
                     resource.data.studentId == request.auth.uid;
      
      // Teachers (approved or lead) can read submissions from their school
      allow read: if isSignedIn() &&
                     isTeacher() &&
                     resource.data.get('schoolId', '') != '' &&
                     isSameSchool(resource.data.schoolId);
      
      // Students can create submissions
      allow create: if isSignedIn() && 
                       request.resource.data.studentId == request.auth.uid;
      
      // Teachers can update submissions (approve/reject)
      allow update: if isSignedIn() &&
                       isTeacher() &&
                       resource.data.get('schoolId', '') != '' &&
                       isSameSchool(resource.data.schoolId);
      
      // Admins can read and update all submissions
      allow read, update: if isAdmin();
      
      // Admins can delete submissions
      allow delete: if isAdmin();
    }
    
    // ─── Activity Feed ──────────────────────────────────────────────
    
    match /ActivityFeed/{activityId} {
      // Users can read activities from their school
      allow read: if isSignedIn() && 
                     (resource.data.get('schoolId', '') == '' || 
                      isSameSchool(resource.data.schoolId));
      
      // Students can create their own activities
      allow create: if isSignedIn() && 
                       request.resource.data.userId == request.auth.uid;
      
      // Admins can read all activities
      allow read: if isAdmin();
    }
    
    // ─── Announcements ──────────────────────────────────────────────
    
    match /Announcements/{announcementId} {
      // Users can read announcements for their school
      allow read: if isSignedIn() && 
                     (resource.data.get('schoolId', '') == '' ||
                      isSameSchool(resource.data.schoolId));
      
      // Approved teachers can create announcements for their school
      allow create: if isSignedIn() && 
                       isApprovedTeacher() &&
                       (request.resource.data.get('schoolId', '') == getCurrentUser().get('schoolId', '') ||
                        request.resource.data.get('schoolId', '') == '');
      
      // Admins can create announcements for all schools
      allow create: if isAdmin();
      
      // Admins can read, update, and delete all announcements
      allow read, update, delete: if isAdmin();
    }
    
    // ─── Push Notifications ─────────────────────────────────────────
    
    match /PushNotifications/{notificationId} {
      // Everyone can read push notifications
      allow read: if isSignedIn();
      
      // Only approved teachers and admins can create push notifications
      allow create: if isApprovedTeacher() || isAdmin();
      
      // Only approved teachers and admins can update push notifications
      allow update: if isApprovedTeacher() || isAdmin();
      
      // Only approved teachers and admins can delete push notifications
      allow delete: if isApprovedTeacher() || isAdmin();
    }
    
    // ─── Platform Settings ──────────────────────────────────────────
    
    match /PlatformSettings/{settingId} {
      // Everyone can read platform settings (needed for daily reminders)
      allow read: if isSignedIn();
      
      // Only approved teachers and admins can write platform settings
      allow write: if isApprovedTeacher() || isAdmin();
    }
    
    // ─── Activity Logs ──────────────────────────────────────────────
    
    match /ActivityLogs/{logId} {
      allow read: if isAdmin();
      allow create: if isSignedIn();
      allow update, delete: if isAdmin();
    }
    
    // ─── Leaderboard ────────────────────────────────────────────────
    
    match /Leaderboard/{entryId} {
      allow read: if isSignedIn();
      allow write: if isSignedIn();
    }
    
    // ─── User Progress & Scores ─────────────────────────────────────
    
    match /UserProgress/{userId} {
      // Users can read and write their own progress
      allow read, write: if isSignedIn() && request.auth.uid == userId;
      
      // Teachers can read progress of students from their school
      allow read: if isSignedIn() && isApprovedTeacher() && 
                     exists(/databases/$(database)/documents/Users/$(userId)) &&
                     get(/databases/$(database)/documents/Users/$(userId)).data.role == "student" &&
                     isSameSchool(get(/databases/$(database)/documents/Users/$(userId)).data.get('schoolId', ''));
      
      // Admins can read all progress
      allow read: if isAdmin();
    }
    
    match /QuizScores/{scoreId} {
      // Users can read and write their own scores
      allow read, write: if isSignedIn() && resource.data.userId == request.auth.uid;
      
      // Teachers can read scores of students from their school
      allow read: if isSignedIn() && isApprovedTeacher() && 
                     exists(/databases/$(database)/documents/Users/$(resource.data.userId)) &&
                     get(/databases/$(database)/documents/Users/$(resource.data.userId)).data.role == "student" &&
                     isSameSchool(get(/databases/$(database)/documents/Users/$(resource.data.userId)).data.get('schoolId', ''));
      
      // Admins can read all scores
      allow read: if isAdmin();
    }
    
    // ─── Admin Users Collection ─────────────────────────────────────
    
    match /AdminUsers/{adminId} {
      // Only admins can read admin users
      allow read: if isAdmin();
      
      // Only admins can create admin users
      allow create: if isAdmin();
      
      // Only admins can update admin users
      allow update: if isAdmin();
      
      // Only admins can delete admin users
      allow delete: if isAdmin();
    }
    
    // ─── Admin Activity Collection ──────────────────────────────────
    
    match /AdminActivity/{activityId} {
      // Only admins can read admin activity logs
      allow read: if isAdmin();
      
      // Only admins can create admin activity logs
      allow create: if isAdmin();
      
      // No updates or deletes allowed (audit trail)
      allow update, delete: if false;
    }
    
    // ─── User Subcollections (completions, quizAttempts) ────────────
    
    match /Users/{userId}/completions/{completionId} {
      // Users can read and write their own completions
      allow read, write: if isSignedIn() && request.auth.uid == userId;
      
      // Teachers can read completions of students from their school
      allow read: if isSignedIn() && isApprovedTeacher() && 
                     exists(/databases/$(database)/documents/Users/$(userId)) &&
                     get(/databases/$(database)/documents/Users/$(userId)).data.role == "student" &&
                     isSameSchool(get(/databases/$(database)/documents/Users/$(userId)).data.get('schoolId', ''));
      
      // Admins can read all completions
      allow read: if isAdmin();
    }
    
    match /Users/{userId}/quizAttempts/{attemptId} {
      // Users can read and write their own quiz attempts
      allow read, write: if isSignedIn() && request.auth.uid == userId;
      
      // Teachers can read quiz attempts of students from their school
      allow read: if isSignedIn() && isApprovedTeacher() && 
                     exists(/databases/$(database)/documents/Users/$(userId)) &&
                     get(/databases/$(database)/documents/Users/$(userId)).data.role == "student" &&
                     isSameSchool(get(/databases/$(database)/documents/Users/$(userId)).data.get('schoolId', ''));
      
      // Admins can read all quiz attempts
      allow read: if isAdmin();
    }
    
    match /Users/{userId}/challengeSubmissions/{submissionId} {
      // Users can read and write their own challenge submissions
      allow read, write: if isSignedIn() && request.auth.uid == userId;
      
      // Teachers can read challenge submissions of students from their school
      allow read: if isSignedIn() && isApprovedTeacher() && 
                     exists(/databases/$(database)/documents/Users/$(userId)) &&
                     get(/databases/$(database)/documents/Users/$(userId)).data.role == "student" &&
                     isSameSchool(get(/databases/$(database)/documents/Users/$(userId)).data.get('schoolId', ''));
      
      // Admins can read all challenge submissions
      allow read: if isAdmin();
    }
    
    // ─── Default Deny All ───────────────────────────────────────────
    
    // Any collection not explicitly defined above is denied
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
